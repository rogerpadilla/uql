```ts
// $between — inclusive range
$where: { age: { $between: [18, 65] } }
// → "age" BETWEEN $1 AND $2

$where: { roles: { $size: 3 } }
// → array length comparison

// $elemMatch — array element matches all conditions
$where: { addresses: { $elemMatch: { city: 'NYC', zip: '10001' } } }

// $not — field-level negation
$where: { status: 'active', name: { $not: { $startsWith: 'test' } } }
// → NOT ("name" LIKE $1)
```

**Location**: Immediately after Task 1 (Comparison Operators), same "###" level.

**Source**: `packages/uql-orm/src/type/query.ts:96-126`, `uql-web/src/content/docs/querying/logical-operators.md`

Add section "### Logical Operators" with examples:

```ts
// $or — match any clause
$where: { $or: [{ name: { $istartsWith: 'A' } }, { status: 'pending' }] }

// $nor — match none (NOT (A OR B))
$where: { $nor: [{ name: 'admin' }, { status: 'banned' }] }

// $not — clause-level negation (NOT (A AND B))
$where: { $not: [{ name: 'admin' }, { status: 'banned' }] }

// Nested — $or with $and inside
$where: {
  $or: [
    { name: { $istartsWith: 'A' } },
    { $and: [{ status: 'pending' }, { createdAt: { $lt: new Date('2025-01-01') } }] }
  ]
}
```

## Task 4: Add Raw SQL Section

**Location**: After Task 3 ("Upsert & Delete"), same level. Create new "### Raw SQL" subsection.

**Source**: `uql-web/src/content/docs/querying/raw-sql.md`, `uql-web/src/content/docs/querying/sub-queries.md`

```ts
// Raw SELECT with generics
const results = await querier.all<{ id: number; total: number }>(
  "SELECT user_id, COUNT(*) AS total FROM orders GROUP BY user_id"
);

// Raw INSERT/UPDATE/DELETE
await querier.run("UPDATE users SET status = $1 WHERE id = $2", ['active', 1]);

// raw() in $where — inject SQL fragments
import { raw } from 'uql-orm';
const items = await querier.findMany(Item, {
  $where: { $and: [{ companyId: 1 }, raw('SUM(salePrice) > 500')] }
});
// → WHERE "companyId" = $1 AND SUM(salePrice) > 500
```

## Task 5: Add Soft-Delete Section

**Location**: After Task 4 ("Raw SQL"), same level. Create new "### Soft-Delete" subsection.

**Source**: `uql-web/src/content/docs/entities/soft-delete.md`

```ts
@Entity({ softDelete: true })
export class User {
  @Id() id?: number;
  @Field() name?: string;
}

// Soft-deleted records are hidden by default
const activeUsers = await querier.findMany(User, { $where: { name: 'alice' } });

// Include deleted with softDelete: true option
const allUsers = await querier.findMany(User, { $where: { name: 'alice' } }, { softDelete: true });

// Permanently delete (bypass soft-delete)
await querier.deleteOneById(User, 1, { softDelete: true });
```

## Task 6: Add Isolation Levels to Transactions

```ts
// Functional API with isolation level
await querier.transaction(async (querier) => {
  const user = await querier.findOne(User, { $where: { email: '...' } });
  await querier.insertOne(Profile, { userId: user.id, bio: '...' });
}, { isolationLevel: 'serializable' });

// Decorator with options
export class UserService {
  @Transactional({ isolationLevel: 'read committed' })
  async register({ picture, ...user }: UserProfile, @InjectQuerier() querier?: Querier) {
    const userId = await querier.insertOne(User, user);
    await querier.insertOne(Profile, { userId, picture });

}
```
